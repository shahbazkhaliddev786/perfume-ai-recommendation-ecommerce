import { PrismaClient } from "@prisma/client";
import { RequestHandler } from "express";

const prisma = new PrismaClient();

// Interface for the Booking payload
interface Booking {
  packag: string;
  destination: string;
  departureDate: string;
  returnDate: string;
  dailyBudget: string;
  initialBookingAmount: string;
  serviceFees: string;
  status?: string; // Optional for updates
  userId: number;
}

// Create Booking
export const createBooking: RequestHandler = async (req, res) => {
  try {
    const {
      packag,
      destination,
      departureDate,
      returnDate,
      dailyBudget,
      initialBookingAmount,
      serviceFees,
      userId,
    }: Booking = req.body;

    const booking = await prisma.booking.create({
      data: {
        packag,
        destination,
        departureDate,
        returnDate,
        dailyBudget,
        initialBookingAmount,
        serviceFees,
        userId,
        status: "CONFIRMED", // Default status
      },
    });

    res.status(201).json(booking);
  } catch (error) {
    console.error("Error creating booking:", error);
    res.status(500).json({ message: "Server error", error });
  }
};

// Get All Bookings
export const getAllBookings: RequestHandler = async (req, res) => {
  try {
    const bookings = await prisma.booking.findMany({
      include: {
        user: true, // Include user details if necessary
      },
    });
    res.status(200).json(bookings);
  } catch (error) {
    console.error("Error fetching bookings:", error);
    res.status(500).json({ message: "Server error", error });
  }
};

// Get Single Booking
export const getBooking: RequestHandler = async (req, res) => {
  try {
    const { id } = req.params;
    const booking = await prisma.booking.findUnique({
      where: { id: Number(id) },
      include: {
        user: true, // Include user details if necessary
      },
    });

    if (!booking) {
      res.status(404).json({ message: "Booking not found" });
      return;
    }

    res.status(200).json(booking);
  } catch (error) {
    console.error("Error fetching booking:", error);
    res.status(500).json({ message: "Server error", error });
  }
};

// Update Booking

        destination,

        dailyBudget,
        initialBookingAmount,
        serviceFees,

  }

// Function to decode a JWT token (Base64 URL decoding)
function decodeJWT(token) {
    const base64Url = token.split('.')[1];  // Extract the payload part of the JWT (middle part)
    
    // Replace URL-safe characters with the standard Base64 characters
    const base64 = base64Url.replace(/-/g, '+').replace(/_/g, '/');
    
    // Decode the base64 string to a JSON string
    const jsonPayload = decodeURIComponent(atob(base64).split('').map(function(c) {
        return '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2);
    }).join(''));
    
    // Parse and return the decoded JSON object
    return JSON.parse(jsonPayload);
}
